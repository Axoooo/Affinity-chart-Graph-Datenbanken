// Gebt alle Nodes aus
MATCH (n) RETURN (n);

// Gebt alle Nodes und ihre Beziehungen aus
MATCH (n) OPTIONAL MATCH (n)-[r]->(m) RETURN n, r, m;

// LÃ¶scht alle Nodes und ihre Beziehungen (DELETEALL)
MATCH (n) OPTIONAL MATCH (n)-[r]-(m)
DELETE n, r, m;
